const backgroundImg = new Image(); backgroundImg.src = "background.png"; let backgroundLoaded = false; backgroundImg.onload = () => { backgroundLoaded = true; }; function drawBackground() { backgroundX -= 1; if (backgroundX <= -canvas.width) backgroundX = 0; if (backgroundLoaded) { ctx.drawImage(backgroundImg, backgroundX, 0, canvas.width, canvas.height); ctx.drawImage(backgroundImg, backgroundX + canvas.width, 0, canvas.width, canvas.height); } else { ctx.fillStyle = "#70c5ce"; ctx.fillRect(0, 0, canvas.width, canvas.height); } }
